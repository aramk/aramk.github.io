JavaScript Module Pattern

My take on the JavaScript module pattern. I needed to call init() when ready. 
(function($) {

	window.moduleName = new function() {
		var base = this;

		// Internally visible
		var pvt = '123';

		// Externally visible
		base.init = function() {
			console.log('init');
		};
	};

})(jQuery);

console.log(moduleName.init);
